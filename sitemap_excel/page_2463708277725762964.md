### `execute`
(
  * `command`

)
#### 参数:
  * `command` String
  * `[arg]` String | Object optional


#### 示例:
```
function main(spreadsheetReport){var command = spreadsheetReport.command;//在新窗口打开电子表格var pageId = spreadsheetReport.getClientId();// 电子表格节点id，如 "I8a8a910a0175e335e335ca510175e3aee2c30806"
    command.execute('OPEN_NEWWINDOW', pageId);//刷新报表
    command.execute('REFRESH');//收藏报表
    command.execute('ADD_FAVORITE');//打印报表
    command.execute('PRINT');//导出相关
    command.execute('EXPORT_HTML');
    command.execute('EXPORT_PNG');
    command.execute('EXPORT_PDF');
    command.execute('EXPORT_WORD');
    command.execute('EXPORT_EXCEL');}
```
